006package org.anarres.gradle.plugin.velocity;
007
008import groovy.lang.Closure;
009import javax.annotation.Nonnull;
010import org.gradle.api.file.SourceDirectorySet;
011import org.gradle.api.internal.file.DefaultSourceDirectorySet;
012import org.gradle.api.internal.file.FileResolver;
013import org.gradle.util.ConfigureUtil;
014
015/**
016 *
017 * @author shevek
018 */
019public class VelocitySourceSet {
020
021    private final SourceDirectorySet velocity;
022
023    public VelocitySourceSet(String displayName, FileResolver fileResolver) {
024        velocity = new DefaultSourceDirectorySet(String.format("Velocity %s source", displayName), fileResolver);
025        velocity.getFilter().include("**/*.java");
026    }
027
028    @Nonnull
029    public SourceDirectorySet getVelocity() {
030        return velocity;
031    }
032
033    @Nonnull
034    public VelocitySourceSet velocity(Closure configureClosure) {
035        ConfigureUtil.configure(configureClosure, getVelocity());
036        return this;
037    }
038}
