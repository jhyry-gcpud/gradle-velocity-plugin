001package org.anarres.gradle.plugin.velocity;
002
003import java.io.File;
004import java.util.ArrayList;
005import java.util.List;
006import java.util.Map;
007import java.util.concurrent.Callable;
008import javax.annotation.Nonnull;
009import javax.inject.Inject;
010import org.gradle.api.Action;
011import org.gradle.api.Plugin;
012import org.gradle.api.Project;
013import org.gradle.api.internal.file.FileResolver;
014import org.gradle.api.internal.plugins.DslObject;
015import org.gradle.api.plugins.JavaPluginConvention;
016import org.gradle.api.tasks.SourceSet;
017
018/**
019 * The velocity plugin.
020 *
021 * This creates a default configuration which preprocesses files
022 * according to the VelocityPluginExtension.
023 *
024 * @author shevek
025 */
026public class VelocityPlugin implements Plugin<Project> {
027
028    private final FileResolver fileResolver;
029
030    @Inject
031    public VelocityPlugin(FileResolver fileResolver) {
032        this.fileResolver = fileResolver;
033    }
034
035    @Override
036    public void apply(final Project project) {
037        final VelocityPluginExtension extension = project.getExtensions().create("velocity", VelocityPluginExtension.class);
038
039        project.getConvention().getPlugin(JavaPluginConvention.class).getSourceSets().all(
040                new Action<SourceSet>() {
041                    @Override
042                    public void execute(SourceSet t) {
043                        apply(project, t, extension);
044                    }
045                });
046    }
047
048    private void apply(@Nonnull final Project project, @Nonnull SourceSet sourceSet, @Nonnull final VelocityPluginExtension extension) {
049        final VelocitySourceSet velocitySourceSet = new VelocitySourceSet(sourceSet.getName(), fileResolver);
050        new DslObject(sourceSet).getConvention().getPlugins().put("velocity", velocitySourceSet);
051        final String srcDir = String.format("src/%s/velocity", sourceSet.getName());
052        velocitySourceSet.getVelocity().srcDir(srcDir);
053        sourceSet.getAllSource().source(velocitySourceSet.getVelocity());
054
055        final String velocityTaskName = sourceSet.getTaskName("process", "Velocity");
056        VelocityTask velocityTask = project.getTasks().create(velocityTaskName, VelocityTask.class, new Action<VelocityTask>() {
057            @Override
058            public void execute(VelocityTask task) {
059                task.setDescription("Preprocesses velocity template files.");
060
061                task.conventionMapping("includeDirs", new Callable<List<File>>() {
062                    @Override
063                    public List<File> call() {
064                        List<Object> includeDirs = extension.includeDirs;
065                        if (includeDirs == null)
066                            return null;
067                        List<File> out = new ArrayList<File>();
068                        for (Object includeDir : includeDirs)
069                            out.add(project.file(includeDir));
070                        return out;
071                    }
072                });
073
074                task.conventionMapping("contextValues", new Callable<Map<String, Object>>() {
075                    @Override
076                    public Map<String, Object> call() {
077                        return extension.contextValues;
078                    }
079                });
080            }
081        });
082
083        velocityTask.setSource(velocitySourceSet.getVelocity());
084
085        final String outputDirectoryName = String.format("%s/generated-sources/antlr/%s",
086                project.getBuildDir(), sourceSet.getName());
087        File outputDir = new File(outputDirectoryName);
088        velocityTask.setOutputDir(outputDir);
089        sourceSet.getJava().srcDir(outputDir);
090
091        project.getTasks().getByName(sourceSet.getCompileJavaTaskName()).dependsOn(velocityTask);
092    }
093}
