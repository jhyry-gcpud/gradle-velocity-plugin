001package org.anarres.gradle.plugin.velocity;
002
003import java.io.File;
004import java.util.ArrayList;
005import java.util.List;
006import java.util.Map;
007import java.util.concurrent.Callable;
008import org.gradle.api.Action;
009import org.gradle.api.Plugin;
010import org.gradle.api.Project;
011import org.gradle.api.Task;
012import org.gradle.api.plugins.JavaPluginConvention;
013import org.gradle.api.tasks.SourceSet;
014import org.gradle.api.tasks.SourceSetContainer;
015
016/**
017 * The velocity plugin.
018 *
019 * This creates a default configuration which preprocesses files
020 * according to the VelocityPluginExtension.
021 *
022 * @author shevek
023 */
024public class VelocityPlugin implements Plugin<Project> {
025
026    @Override
027    public void apply(final Project project) {
028        final VelocityPluginExtension extension = project.getExtensions().create("velocity", VelocityPluginExtension.class);
029        Task velocityVppTask = project.getTasks().create("velocityVpp", VelocityTask.class, new Action<VelocityTask>() {
030            @Override
031            public void execute(VelocityTask task) {
032                task.setDescription("Preprocesses velocity template files.");
033
034                task.conventionMapping("inputDir", new Callable<File>() {
035                    @Override
036                    public File call() {
037                        return project.file(extension.inputDir);
038                    }
039                });
040
041                task.conventionMapping("includeDirs", new Callable<List<File>>() {
042                    @Override
043                    public List<File> call() {
044                        List<Object> includeDirs = extension.includeDirs;
045                        if (includeDirs == null)
046                            return null;
047                        List<File> out = new ArrayList<File>();
048                        for (Object includeDir : includeDirs)
049                            out.add(project.file(includeDir));
050                        return out;
051                    }
052                });
053
054                task.conventionMapping("outputDir", new Callable<File>() {
055                    @Override
056                    public File call() {
057                        return project.file(extension.outputDir);
058                    }
059                });
060
061                task.conventionMapping("contextValues", new Callable<Map<String, Object>>() {
062                    @Override
063                    public Map<String, Object> call() {
064                        return extension.contextValues;
065                    }
066                });
067            }
068        });
069
070        project.getTasks().getByName("compileJava").dependsOn(velocityVppTask);
071        SourceSetContainer sourceSets = project.getConvention().getPlugin(JavaPluginConvention.class).getSourceSets();
072        final SourceSet mainSourceSet = sourceSets.getByName("main");
073        mainSourceSet.getJava().srcDir(extension.outputDir);
074    }
075
076}
